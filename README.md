    }

    class FullTimeEmployee : BaseEmployee
    {
        public override double CalculateSalary(int workingHours)

        }

    class ContractEmployee : BaseEmployee

        public override double CalculateSalary(int workingHours)
        {

Интерфейси (100% абстракция)
